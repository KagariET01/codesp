namespace ET01{
	template<typename T>INT LCS(T af,T as,T bf, T bs){
		INT n=as-af;
		INT m=bs-bf;
		INT dp[n+1][m+1]={};
		T it1=af;
		T it2=bf;
		for(INT i=1;i<=n;i++){
			it2=bf;
			for(INT j=1;j<=m;j++){
				if(*it1==*it2)dp[i][j]=dp[i-1][j-1]+1;
				else dp[i][j]=dp[i-1][j];
				it2++;
			}
			it1++;
		}
		return dp[n][m];
	}
}
